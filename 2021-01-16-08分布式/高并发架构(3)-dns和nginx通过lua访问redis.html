    1、域名、DNS

    当流量极大，不能用某一台单独的机器来接入流量。nginx 之能 5 万。即使是物理的 F4 负载均衡，只要在一台服务器上，肯定是有上限的。所以需要通过域名来配置多个服务器。

    1.1 浏览器如何通过域名得到 ip

    1. 查询本地 hosts 文件
    2. 查询本机 dns 缓存
    3. 查询远程 dns 服务器
    4. 把 ip 缓存到本机

    1.2 dns 服务器上为啥有你的域名和 ip 映射？

    你购买域名时，需要配置解析，即域名对应的服务器，并且需要配置 dns 服务器。dns 服务器会互相数据广播，把你配置的解析同步到网上的 dns 服务器。

    1.3 dns 服务器支持域名、ip一对多

    配置多个 ip 目的是为了达到基于域名的负载均衡。

    但是需要一定的业务逻辑，让 dns 服务器知道，到底该给你哪个 ip。这个业务逻辑的实现需要在配置域名解析时的 dns 服务器是智能 dns，智能 dns 可以获取客户端请求的 ip 地址，根据 ip 地址分析地区或者 ISP 供应商，然后返回对应地区的 ip 地址，即返回离客户端最近的 ip 地址。

    1.4 CDN 节点组成

    • 负载均衡设备
    • 高速缓存服务器

    其实就是一个负载均衡器 lvs 或 nginx，配合 keepalived 做到入口高可用，代理或转发请求到内网节点做负载均衡。

    1.5 CDN 应用场景

    适用

    1. css、js、img 等 html 骨架中包含的静态资源
    2. app 内需要的 img、html 等资源
    3. 音视频点播资源
    4. 大文件下载

    不适用

    1. 私密的商业数据
    2. 包含有逻辑的请求
    3. 需要鉴权
    4. 长连接
    5. reactor 模型
    6. 即时通讯

    2、Nginx 访问 redis

    2.1 redis2-nginx-module

    redis2-nginx-module 是一个支持 Redis 2.0 协议的 Nginx upstream 模块，它可以让 Nginx 以非阻塞方式直接防问远方的 Redis 服务，同时支持 TCP 协议和 Unix Domain Socket 模式，并且可以启用强大的 Redis 连接池功能。

    https://github.com/openresty/redis2-nginx-module

    测试

    # redis2-nginx-module 访问 redis，测试
location /redis2_test {
    default_type text/html;
    redis2_pass 127.0.0.1:6379;
    #redis2_query auth pass9876;
    set $value 'first';
    redis2_query set one $value;
}

    set

    # redis2-nginx-module 访问 redis, set
location /redis2_set {
    default_type text/html;
    redis2_pass 127.0.0.1:6379;
    #redis2_query auth pass9876;

    set_unescape_uri $key $arg_key;
    set_unescape_uri $val $arg_val;
    redis2_query set $key $val;
}

    get

    # redis2-nginx-module 访问 redis, get
location /redis2_get {
    default_type text/html;
    redis2_pass 127.0.0.1:6379;
    #redis2_query auth pass9876;

    set_unescape_uri $key $arg_key;
    redis2_query get $key;
}

    pipelined

    # redis2-nginx-module 访问 redis, pipelined
location /redis2_pipeline {
    default_type text/html;
    redis2_pass 127.0.0.1:6379;
    #redis2_query auth pass9876;
    set $value 'fitst';
    redis2_query set one $value;
    redis2_query get one;
    redis2_query set one two;
    redis2_query get one;
}

    list

    # redis2-nginx-module 访问 redis, list
location /redis2_list {
    default_type text/html;
    redis2_pass 127.0.0.1:6379;

    redis2_query lpush key1 C;
    redis2_query lpush key1 B;

    redis2_query lrange key1 0 -1;

    cluster

    # upstream 和 server 同一个层级
upstream redis_cluster {
    server 127.0.0.1:6379;
    server 127.0.0.1:6379;
}


# redis2-nginx-module 访问 redis, cluster
location /redis2_cluster {
    default_type text/html;
    redis2_pass redis_cluster;
    redis2_next_upstream error timeout invalid_response;
    redis2_query get one;
}

    2.2 lua-resty-redis 访问 redis

    2.1 方式访问 redis，不好写业务逻辑，可以用 lua-resty-reids，通过 lua 来直接访问 redis。

    https://github.com/openresty/lua-resty-redis

    nginx.conf

    # lua-resty-redis
location /lua_resty_redis {
    default_type text/html;
    content_by_lua_file conf/lua/lua_resty_redis.lua;
}

    lua_resty_redis.lua

    -- 引入 resty.redis
local redis = require "resty.redis"
-- 初始化实力
local red = redis:new()
red:set_timeouts(1000, 1000, 1000)

local ok, err = red:connect("127.0.0.1", 6379)
if not ok then
    ngx.say("failed to connect:", err)
    return
end

--local res, err = red:auth("pass9876")

--    return

ok, err = red:set("dog", "an animal")
if not ok then
    ngx.say("failed to set dog:", err)
    return
end

ngx.say("set result:", ok)

local res, err = red:get("dog")
if not res then
    ngx.say("failed to get dog: ", err)
    return
end

if res == ngx.null then
    ngx.say("dog not found.")
    return
end

ngx.say("dog: ", res)

red:init_pipeline()
red:set("cat", "Marry")
red:set("horse", "Bob")
red:get("cat")
red:get("horse")

if not results then

        else
            -- process the table value
        end
    else
        -- process the scalar value
    end
end

-- put it into the connection pool of size 100,
-- with 10 seconds max idle time

if not ok then
    ngx.say("failed to set keepalive: ", err)
    return

    3、URL 一致性哈希负载均衡 lua-resty-http

    有针对性的对 url 进行一致性 hash 定向负载到后端 Nginx ，提高 Nginx 缓存系统命中率。

    3.1 nginx url_hash

    Nginx第三方模块，在转发请求时如果后端服务器宕机，会导致503错误

    3.2 lua-resty-http

    https://github.com/ledgetech/lua-resty-http

    安装组件

    wget https://github.com/ledgetech/lua-resty-http/archive/v0.15.tar.gz

    把 http_headers.lua、http.lua 移动到 /usr/local/openresty/lualib/resty 即可。

    发送请求

    nginx.conf

    # lua-resty-http
location /lua_resty_http {
    resolver 8.8.8.8;
    default_type text/html;
    content_by_lua_file conf/lua/lua_resty_http_test.lua;
}

    lua_resty_http_test.lua

    local http = require("resty.http")
local httpc = http.new()
local resp, err = httpc:request_uri("http://www.sogou.com", {
    method = "GET",
    path= "/sogou?query=resty.http",
    headers = {
        ["User-Agent"] = "Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/40.0.2214.111 Safari/537.36"
    },
    keepalive_timeout = 60000,
    keepalive_pool = 10
})

if not resp then
    ngx.say("request error:", err)
    return
end

ngx.status = resp.status

for k,v in pairs(resp.headers) do
    if k ~= "Transfer-Encoding" and k ~= "Connection" then
        ngx.header[k] = v
    end
end

ngx.say(resp.body)
httpc:close()

    实现一致性哈希负载均衡

    nginx.conf

    # lua-resty-http 一致性哈希负载均衡
location /lua_resty_http_lb {
    resolver 8.8.8.8;
    default_type text/html;
    content_by_lua_file conf/lua/lua_resty_http_lb.lua;
}

    lua_resty_http_lb.lua

    local http = require("resty.http")
local httpc = http.new()

local hosts = {"192.168.47.12", "192.168.47.13"}

-- 取请求参数 id，做哈希
local item_id = ngx.var.arg_id

local id_hash = ngx.crc32_long(item_id)
local index = (id_hash % 2) + 1

local resp, err = httpc:request_uri("http://"..hosts[index], {
    method = "GET",
    path = "/?query=resty.http",
    headers = {
        ["User-Agent"] = "Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/40.0.2214.111 Safari/537.36"
    }
})

if not resp then
    ngx.say("request error:", err)
    return
end

ngx.say(resp.body)
httpc:close()

    4、模板实时渲染 lua-resty-template

    https://github.com/bungle/lua-resty-template

    如果学习过 JavaEE 中的 servlet 和 JSP 的话，应该知道 JSP 模板最终会被翻译成 Servlet 来执行；

    而 lua-resty-template 模板引擎可以认为是 JSP，其最终会被翻译成 Lua 代码，然后通过 ngx.print 输出。

    lua-resty-template 大体内容有：

    l 模板位置：从哪里查找模板；

    l 变量输出/转义：变量值输出；

    l 代码片段：执行代码片段，完成如 if/else、for 等复杂逻辑，调用对象函数/方法；

    l 注释：解释代码片段含义；

    l include：包含另一个模板片段；

    l 其他：lua-resty-template还提供了不需要解析片段、简单布局、可复用的代码块、宏指令等支持。

    基础语法

    {(include_file)}：包含另一个模板文件；
{* var *}：变量输出；
{{ var }}：变量转义输出；
{% code %}：代码片段；
{# comment #}：注释；
{-raw-}：中间的内容不会解析，作为纯文本输出；

    4.1 lua 代码热加载

    在 http 模块中加入

    lua_code_cache off;

    reload 后 Nginx 会提示影响性能，记得在生产环境中关掉。
