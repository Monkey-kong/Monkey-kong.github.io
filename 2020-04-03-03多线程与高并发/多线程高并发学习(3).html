    1. 面试题1 - 实现同步容器

    实现一个容器，提供两个方法，add，size，写两个线程，线程1添加10个元素到容器中，线程2实现监控元素的个数，当个数到5个时，线程2给出提示并结束

    1.1 volatile、synchronizedList 实现（不可行）

    public class T01_Interview1 {
    // volatile 保证线程可见。其实这里有问题，volatile 修饰的是引用，应该是检测不到
    // synchronizedList 构建线程安全的 list
    volatile List<Object> list = Collections.synchronizedList(new ArrayList<>());
    void addElement(Object o){
        // 当元素加进去之后，还没来得及 size++，其他线程来读取了。导致读取到的 size 和实际元素不一致
        // 所以 list 需要修改为线程安全的 list
        list.add(o);
    }
    int elementSize(){
        return list.size();
    }
    public static void main(String[] args) {
        T01_Interview1 t = new T01_Interview1();
        Thread t1 = new Thread(() -> {
            System.out.println("t1 start");
            for (int i = 0; i < 10; i++) {
                System.out.println("add " + i);
                t.addElement(new Object());
            }
            System.out.println("t1 end");
        });
        Thread t2 = new Thread(() -> {
            System.out.println("t2 start");
            while (true) {
                // 有时候不准，是因为没有抢占到cpu资源？
                if(t.elementSize() == 5){
                    System.out.println("list size is 5.....");
                    break;
                }
            }
        });
        t1.start();
        t2.start();
    }
}

    1.2 synchronized、wait、notify 实现

    public class T01_Interview2 {
    List<Object> list = new ArrayList<>();
    void addElement(Object o){
        list.add(o);
    }
    int elementSize(){
        return list.size();
    }
    public static void main(String[] args) throws InterruptedException {
        T01_Interview2 t = new T01_Interview2();
        final Object lock = new Object();
        Thread t1 = new Thread(() -> {
            synchronized (lock){
                System.out.println("t1 start");
                for (int i = 0; i < 10; i++) {
                    System.out.println("add " + i);
                    t.addElement(new Object());
                    if(t.elementSize() == 5) {
                        // notify 不会释放锁
                        lock.notify();
                        try {
                            // 释放锁，让 t2 得以执行
                            lock.wait();
                        } catch (InterruptedException e) {
                            e.printStackTrace();
                        }
                    }
                }
                System.out.println("t1 end");
            }
        });
        Thread t2 = new Thread(() -> {
            synchronized (lock){
                try {
                    System.out.println("t2 start");
                    lock.wait();
                    System.out.println("list size is 5.....");
                    lock.notify();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        });
        t2.start();
        // 必须保证 t2 先执行。因为必须先 wait 再 notify，否则 wait 将继续阻塞
        TimeUnit.MILLISECONDS.sleep(200);
        t1.start();
    }
}

    1.3 CountDownLatch 实现

    public class T01_Interview3 {
    List<Object> list = new ArrayList<>();
    void addElement(Object o){
        list.add(o);
    }
    int elementSize(){
        return list.size();
    }
    public static void main(String[] args) throws InterruptedException {
        T01_Interview3 t = new T01_Interview3();
        CountDownLatch ct = new CountDownLatch(1);
        CountDownLatch ct1 = new CountDownLatch(1);

        Thread t1 = new Thread(() -> {
            System.out.println("t1 start");
            for (int i = 0; i < 10; i++) {
                System.out.println("add " + i);
                t.addElement(new Object());
                if(t.elementSize() == 5) {
                    try {
                        ct.countDown();
                        ct1.await();
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
            }
            System.out.println("t1 end");
        });
        Thread t2 = new Thread(() -> {
            try {
                System.out.println("t2 start");
                ct.await();
                System.out.println("list size is 5.....");
                ct1.countDown();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        });
        t1.start();
        t2.start();
    }
}

    1.4 LockSupport +join实现

    public class T01_Interview4 {
    List<Object> list = new ArrayList<>();
    void addElement(Object o){
        list.add(o);
    }
    int elementSize(){
        return list.size();
    }
    public static void main(String[] args) throws InterruptedException {
        T01_Interview4 t = new T01_Interview4();

        Thread t2 = new Thread(() -> {
            System.out.println("t2 start");
            try {
                // 可以先 unpark 再 park，但是必须 unpark 时 t2 已经启动
                TimeUnit.SECONDS.sleep(5);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            System.out.println("start park");
            LockSupport.park();
            System.out.println("list size is 5.....");
        });
        Thread t1 = new Thread(() -> {
            System.out.println("t1 start");
            for (int i = 0; i < 10; i++) {
                System.out.println("add " + i);
                t.addElement(new Object());
                if(t.elementSize() == 5) {
                    LockSupport.unpark(t2);
                    System.out.println("unpark finish...");
                    try {
                        // 这里也可以和 CountDownLunch 类似，用两套
                        t2.join();
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
            }
            System.out.println("t1 end");
        });
        t1.start();
        t2.start();
    }
}

    1.5 双重 LockSupport

    public class T01_Interview5 {
    List<Object> list = new ArrayList<>();
    void addElement(Object o){
        list.add(o);
    }
    int elementSize(){
        return list.size();
    }
    static Thread t1 = null;
    static Thread t2 = null;
    public static void main(String[] args) throws InterruptedException {
        T01_Interview5 t = new T01_Interview5();
        t1 = new Thread(() -> {
            System.out.println("t1 start");
            for (int i = 0; i < 10; i++) {
                System.out.println("add " + i);
                t.addElement(new Object());
                if(t.elementSize() == 5) {
                    LockSupport.unpark(t2);
                    LockSupport.park();
                }
            }
            System.out.println("t1 end");
        });
        t2 = new Thread(() -> {
            System.out.println("t2 start");
            LockSupport.park();
            System.out.println("list size is 5.....");
            LockSupport.unpark(t1);
        });
        t1.start();
        // 有问题。LockSupport unpark 时必须要线程已经 start
        // TimeUnit.SECONDS.sleep(1);
        t2.start();
    }
}

    2. 面试题2 - 实现同步容器2

    面试题：写一个固定容量同步容器，拥有put和get方法，以及getCount方法，能够支持2个生产者线程以及10个消费者线程的阻塞调用。

    2.1 synchronized、wait 和 notifyAll 实现

    public class T02_Interview1<T> {
    final private LinkedList<T> lists = new LinkedList<>();
    final private int MAX = 10;
    private int count = 0;

    public synchronized void put(T t) {
        while(getCount() == MAX){
            System.out.println("list is full....wait");
            try {
                this.wait();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        lists.add(t);
        count++;
        // 通知消费者消费
        this.notifyAll();
    }
    public synchronized T get() {
        // 这里应该用 while 而不是 if。因为都是 notifyAll，只生产两个数据，但是唤醒了 10 个线程
        // 所以每个线程唤醒后，需要重新判断 count 是不是 0
        while(getCount() == 0){
            System.out.println("list is empty....wait");
            try {
                this.wait();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        T t = lists.removeFirst();
        count--;
        // 通知生产者生产
        this.notifyAll();
        return t;
    }
    public synchronized int getCount(){
        return count;
    }

    public static void main(String[] args) {
        T02_Interview1<String> t = new T02_Interview1<>();
        // 消费
        for (int i = 0; i <10; i++) {
            new Thread(()->{
                System.out.println(Thread.currentThread().getName() + " consumer： " + t.get());
            },"c"+i).start();
        }
        // 生产
        for (int i = 0; i < 2; i++) {
            new Thread(()->{
                t.put(Thread.currentThread().getName()+ " data");
            }, "p"+i).start();
        }
    }
}

    2.2 ReentrantLock、Condition 实现

    public class T02_Interview2<T> {
    final private LinkedList<T> lists = new LinkedList<>();
    final private int MAX = 10;
    private int count = 0;
    private ReentrantLock lock = new ReentrantLock();
    private Condition producer = lock.newCondition();
    private Condition consumer = lock.newCondition();

    public void put(T t) {
        try {
            lock.lock();
            while(getCount() == MAX){
                producer.await();
                System.out.println("list is full....wait");
            }
            lists.add(t);
            count++;
            consumer.signalAll();
        } catch (InterruptedException e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }
    public T get() {
        T t = null;
        try {
            lock.lock();
            // 这里应该用 while 而不是 if。因为都是 notifyAll，只生产两个数据，但是唤醒了 10 个线程
            // 所以每个线程唤醒后，需要重新判断 count 是不是 0
            while(getCount() == 0){
                consumer.await();
                System.out.println("list is empty....wait");
            }
            t = lists.removeFirst();
            count--;
            producer.signalAll();
        } catch (InterruptedException e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
        return t;
    }
    public int getCount(){
        return count;
    }
}

    3.3 VarHandle

    指向某个变量的引用。x=x+10;不是原子性的，如果要做到线程安全，需要加锁，但是如果使用 VarHandle 可以不加锁来实现原子操作。所以，VarHandle 可以实现普通属性也可以进行原子操作。

    VarHandle 使用 c、c++ 语言调用 cpu 原语实现。它比反射的效率更高，反射每次都需要检查；VarHandle 可以理解为直接操作二进制码。

    public class T03_VarHandle {
    int x = 8;
    private static VarHandle handle;
    static {
        try {
            // x 可以操作 8，handle 也可以
            handle = MethodHandles.lookup().findVarHandle(T03_VarHandle.class, "x", int.class);
        } catch (NoSuchFieldException e) {
            e.printStackTrace();
        } catch (IllegalAccessException e) {
            e.printStackTrace();
        }
    }
    public static void main(String[] args) {
        T03_VarHandle t = new T03_VarHandle();
        // 获取和设置
        System.out.println(handle.get(t));
        handle.set(t,9);
        System.out.println(t.x);
        // cas操作
        handle.compareAndSet(t, 9, 10);
        System.out.println(t.x);
        // cas操作
        handle.getAndAdd(t, 10);
        System.out.println(t.x);
    }
 }
