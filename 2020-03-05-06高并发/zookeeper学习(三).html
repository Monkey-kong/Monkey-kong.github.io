    ZooKeeper 典型应用场景：分布式配置、分布式锁。

    1. 分布式配置

    1.1 分布式配置分析

    传统的配置一般放在本地，但是当服务集群大了，修改一个配置需要登录多台主机，所以如果有一个固定的位置存放一份统一的配置，只需要改一个配置，所有服务都能生效那是非常好的。这个位置可以是本地文件系统、数据库、Redis 等等，但是如果使用数据库或者 Redis 都需要轮询去取或者手动调用触发，因为不知道配置是否有改动，如果使用 ZooKeeper 的 watch 机制，就可以完美的解决该问题。

    1.2 具体实现

    1. 验证配置节点是否存在
    2. 配置节点存在则获取配置信息
    3. 成功获取配置信息则将信息设置到配置对象，然后放心
    4. 配置节点新增、修改，重新获取配置
    5. 配置节点删除，设置配置对象为空，重新监听配置
    public class AsyncConfigTest {

    private ZooKeeper zk;
    private MyConfig myConfig;
    ZKConfigCompont zkConfigCompont;

    @Before
    public void connect(){
        String connectString = "192.168.238.66:2181,192.168.238.68:2181,192.168.238.70:2181,192.168.238.72:2181/testNode";
        int sessionTimeOut = 10000;
        ZKConfig zkConfig = ZKUtile.initZKConfig(connectString, sessionTimeOut);

        zk = ZKUtile.getInstance(zkConfig);
        myConfig = new MyConfig();

         zkConfigCompont = new ZKConfigCompont("/configTest", zk, myConfig);
        // 阻塞，zk 服务端处理获取配置请求
        zkConfigCompont.watchConfig();
    }

    @After
    public void disconnect(){
        try {
            zk.close();
            System.out.println("closed ....");
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    @Test
    public void test(){
        // 获取配置（如果有更新或者删除响应式实时更新）
        while (true){
            if(myConfig.getConfig() == null || "".equals(myConfig.getConfig())){
                // 配置被删除，重新阻塞
                zkConfigCompont.watchConfig();
            } else {
                System.out.println("当前配置：" + myConfig.getConfig());
            }
            try {
                Thread.sleep(1000l);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}
    public class ZKConfigCompont implements Watcher, AsyncCallback.StatCallback, AsyncCallback.DataCallback {

    private String configPath;
    private ZooKeeper zk;
    private MyConfig myConfig;
    private CountDownLatch cd = new CountDownLatch(1);

    public ZKConfigCompont(String configPath, ZooKeeper zk, MyConfig myConfig){
        this.configPath = configPath;
        this.zk = zk;
        this.myConfig = myConfig;
    }

    public void watchConfig(){
        // 验证配置节点是否存在
        zk.exists(configPath, this, this, "exists");
        try {
            // 阻塞，等待获取到配置信息放行
            cd.await();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    // exists callback
    @Override
    public void processResult(int rc, String path, Object ctx, Stat stat) {
        System.out.println("exists callback：" + rc);
        if (KeeperException.Code.OK.intValue() == rc){
            // 配置节点存在则获取配置信息
            zk.getData(configPath, this, this, "get data");
        }
    }

    // get data callback
    @Override
    public void processResult(int rc, String path, Object ctx, byte[] data, Stat stat) {
        if (KeeperException.Code.OK.intValue() == rc){
            // 设置配置，放行
            myConfig.setConfig(new String(data));
            cd.countDown();
        }
    }

    // exists/getData watch
    @Override
    public void process(WatchedEvent event) {
        Event.EventType type = event.getType();
        // 配置节点新增，重新获取配置
        // 配置节点修改，重新获取配置
        // 配置节点删除，设置配置为空,重新监听配置
        switch (type) {
            case NodeCreated:
            case NodeDataChanged:
                System.out.println("config node create/changed....");
                zk.getData(configPath, this, this, "get data");
                break;
            case NodeDeleted:
                System.out.println("config node deleted....");
                cd = new CountDownLatch(1);
                myConfig.setConfig("");
                // watchConfig(); // 这里不知为何不能这么处理，待研究
                break;
        }
    }
}
    public class ZKUtile {

    public static ZooKeeper getInstance(ZKConfig zkConfig){
        CountDownLatch cd = zkConfig.getCd();
        ZooKeeper zk = null;
        try {
            zk = new ZooKeeper(zkConfig.getConnectString(), zkConfig.getSessionTimeout(), zkConfig.getWatcher());
            cd.await();
        } catch (Exception e) {
            System.out.println("get zk instance failed...");
            e.printStackTrace();
        }
        return zk;
    }

    public static ZKConfig initZKConfig(String connectString, int sessionTimeout){
        ZKConfig zkConfig = new ZKConfig();
        CountDownLatch cd = new CountDownLatch(1);
        zkConfig.setConnectString(connectString);
        zkConfig.setSessionTimeout(sessionTimeout);
        zkConfig.setCd(cd);
        zkConfig.setWatcher(new Watcher() {
            // 父节点 watch 事件
            @Override
            public void process(WatchedEvent event) {
                Event.EventType type = event.getType();

                Event.KeeperState state = event.getState();
                switch (state) {
                    case Disconnected:

                    case SyncConnected:
                        // 节点创建，放行
                        System.out.println(event.getPath()+"  connected....");
                        cd.countDown();
                        break;
                    case AuthFailed:
                        break;
                    case ConnectedReadOnly:
                        break;
                    case SaslAuthenticated:
                        break;
                    case Expired:
                        break;
                }
            }
        });
        return zkConfig;
    }
}

    2. 分布式锁

    2.1 分布式锁分析

    1. 多个客户端争抢锁，只能有一个人获得锁
    2. 如果获得锁的人出问题了，利用临时节点和 session 机制，防止死锁
    3. 如果获得锁的人成功了，释放锁
    4. 其他客户端如何知道锁被释放了？
      1. 客户端主动轮询，心跳，查看锁节点是否存在。问题：存在延迟；zk 服务端压力大，多个客户端不停仿佛服务端。
      2. watch：解决延迟问题。问题：多个客户端 watch 同一个锁节点，当锁释放时，zk 服务端需要同时给大量 watch 了该节点的客户端回调，zk 服务端压力大
      3. sequence+watch：sequence 形式创建节点，每个客户端 watch 前一个客户端的锁节点。一旦最小的释放了锁，zk 只给第二个客户端发事件回调。

    2.2 具体实现

    1. 创建分布式锁工作根目录：/testLock
    2. 想要获取锁的线程先在根目录下创建 EPHEMERAL_SEQUENTIAL 节点
    3. 创建完成后获取根目录下所有节点
    4. 根目录下所有子节点排序后判断自己是不是第一个
    5. 如果是第一个获得锁，放行
    6. 如果不是第一个，watch 排序后该节点的前一个节点的 NodeDeleted 事件，线程阻塞
    7. 当前获得锁的节点工作完成释放锁，触发 NodeDeleted 事件，watch 它的节点开始枪锁
    public class ZKLockComponent implements AsyncCallback.StringCallback, AsyncCallback.Children2Callback, Watcher {

    private ZooKeeper zk;
    private String lockNamePrex; // 锁名称前缀（/lock）
    private String lockName;     // 锁节点名称（/lock0000000080 /lock0000000081 ...）
    private String threadName;   // 工作线程名称 （Thread-0 Thread-1 ...）
    private CountDownLatch cd = new CountDownLatch(1);

    public ZKLockComponent(ZooKeeper zk, String lockNamePrex, String threadName){
        this.zk = zk;
        this.lockNamePrex = lockNamePrex;
        this.threadName = threadName;
    }

    public void tryLock(){
        // 新增锁节点
        zk.create(lockNamePrex, threadName.getBytes(), ZooDefs.Ids.OPEN_ACL_UNSAFE,
                CreateMode.EPHEMERAL_SEQUENTIAL, this, "aaa");
        try {
            cd.await();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    // create callback
    @Override
    public void processResult(int rc, String path, Object ctx, String name) {
        System.out.println("path:" + path + "，threadName:" + threadName + "，name：" + name);
        lockName = name;
        if(KeeperException.Code.OK.intValue() == rc){
            // 根节点无需 watch
            zk.getChildren("/",null, this, "aaa");
        }
    }

    // getChildren callback
    @Override
    public void processResult(int rc, String path, Object ctx, List<String> children, Stat stat) {
        if(children != null){
            try {
                Collections.sort(children);
                int i = children.indexOf(lockName.substring(1));
                if(i == 0){
                    // 判断自己是不是最小的，是最小的就放行获得锁
                    System.out.println(threadName+" i am first...");
                    // 设置锁根节点的值为当前获得锁的线程名称
                    zk.setData("/",threadName.getBytes(),-1);
                    cd.countDown();
                } else {
                    // 不是最小的，watch 前面的一个
                    System.out.println(threadName+" watch "+children.get(i-1));
                    zk.exists("/"+children.get(i-1),this);
                }
            } catch (KeeperException e) {
                e.printStackTrace();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }

    // exists watch
    @Override
    public void process(WatchedEvent event) {
        // 前面节点删除，立即抢锁
        Event.EventType type = event.getType();
        switch (type) {
            case NodeCreated:
                break;
            case NodeDeleted:
                zk.getChildren("/",null, this, "");
                break;
        }
    }

    public void unLock(){
        try {
            zk.delete(lockName,-1);
        } catch (InterruptedException e) {
            e.printStackTrace();
        } catch (KeeperException e) {
            e.printStackTrace();
        }
    }

    public void getRootData() throws KeeperException, InterruptedException {
        byte[] data = zk.getData("/", false, new Stat());
        System.out.println("RootData："+new String(data));
    }
}
