    姓名 性别 年龄

    select max(case s.type when 1 then s.value end) '姓名',
  max(case s.type when 2 then s.value end) '性别',
  max(case s.type when 3 then s.value end) '年龄'
from student s
group by s.t_id
;

    create table tmp(rq varchar(20),shengfu varchar(20));

insert into tmp values('2005-05-09','胜');
insert into tmp values('2005-05-09','胜');
insert into tmp values('2005-05-09','负');
insert into tmp values('2005-05-09','负');

    如果要生成下列结果, 该如何写 sql 语句?

    日期 胜场数 负场数
    2005-05-09 2 2
    2005-05-10 1 2 
    select rq '日期', 
  sum(case shengfu when '胜' then 1 else 0 end) '胜场数',
  sum(case shengfu when '胜' then 0 else 1 end) '负场数' 
  FROM tmp
    group by rq
;

    1、 case…when….then + group by + sum 行转列

    select userid, 
  sum(case subject when '语文' then score else 0 end) '语文',
  sum(case subject when '数学' then score else 0 end) '数学',
  sum(case subject when '英语' then score else 0 end) '英语',
  sum(case subject when '政治' then score else 0 end) '政治'
from tb_score
group by userid
;

    2、 IF() + group by + sum 行转列：

    select userid,
  sum(if(subject='语文',score,0)) '语文',
  sum(if(subject='数学',score,0)) '数学',
  sum(if(subject='英语',score,0)) '英语',
  sum(if(subject='政治',score,0)) '政治'
from tb_score
group by userid
;

    4、 SUM(IF()) 生成列 + UNION 生成汇总行

    select userid,
  sum(if(subject='语文',score,0)) '语文',
  sum(if(subject='数学',score,0)) '数学',
  sum(if(subject='英语',score,0)) '英语',
  sum(if(subject='政治',score,0)) '政治',
    sum(score) '总分'
from tb_score
group by userid
union
select 'Total',sum(if(subject='语文',score,0)) '语文',
  sum(if(subject='数学',score,0)) '数学',
  sum(if(subject='英语',score,0)) '英语',
  sum(if(subject='政治',score,0)) '政治',
  sum(score)
from tb_score
;

    5、 SUM(IF()) 生成列 + WITH ROLLUP 生成汇总行

    select IFNULL(userid,'Total') userid,
  sum(if(subject='语文',score,0)) '语文',
  sum(if(subject='数学',score,0)) '数学',
  sum(if(subject='英语',score,0)) '英语',
  sum(if(subject='政治',score,0)) '政治',
    sum(score) '总分'
from tb_score
group by userid
with rollup
;

    6、 动态，适用于列不确定情况

    思路就是拼接 sql 语句然后用 **PREPARE name from sqlStr、EXECUTE name 、DEALLOCATE PREPARE name ** 预处理、执行、回收资源。

    SET @EE='';
select @EE :=CONCAT(@EE,'sum(if(subject= \'',subject,'\',score,0)) as ',subject, ',') AS aa FROM (SELECT DISTINCT subject FROM tb_score) A ;
-- select @EE;

SET @QQ = CONCAT('select ifnull(userid,\'TOTAL\')as userid,',@EE,' sum(score) as TOTAL from tb_score group by userid WITH ROLLUP');
-- SELECT @QQ;

PREPARE stmt FROM @QQ;
EXECUTE stmt;
DEALLOCATE PREPARE stmt;

    7、 合并字段显示：利用group_concat()

    SELECT userid,GROUP_CONCAT(`subject`,":",score)AS 成绩 
FROM tb_score
GROUP BY userid
;

    8、列转行

    CREATE TABLE tb_score1(
    id INT(11) NOT NULL auto_increment,

SELECT userid,'语文' AS course,cn_score AS score FROM tb_score1
union all
