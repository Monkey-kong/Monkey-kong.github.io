    2. 如何启动线程？

    /**
 * 创建线程的方式
 * 1. 继承 Thread
 * 2. 实现 Runnable
 * 3. lambda(其实就是 Runnable)
 * 4. FutureTask
 */
public class T01_CreateThread {
    static class ThreadDemo01 extends Thread{
        @Override
        public void run() {
            for (int i = 0; i < 10000; i++) {
                System.out.println(Thread.currentThread().getName() + "=====" + i);
            }
        }
    }
    static class ThreadDemo02 implements Runnable{
        @Override
        public void run() {
            for (int i = 0; i < 10000; i++) {
                System.out.println(Thread.currentThread().getName() + "=====" + i);
            }
        }
    }
    public static void main(String[] args) throws ExecutionException, InterruptedException {
        new Thread(new ThreadDemo01(), "ThreadDemo01").start();
        new Thread(new ThreadDemo02(), "ThreadDemo02").start();
        new Thread(()->{
            for (int i = 0; i < 10000; i++) {
                System.out.println(Thread.currentThread().getName() + "=====" + i);
            }
        }, "ThreadDemo03").start();

        FutureTask<Integer> task = new FutureTask<Integer>(()->{
            for (int i = 0; i < 10000; i++) {
                System.out.println(Thread.currentThread().getName() + "=====" + i);
            }
            return 100;
        });
        new Thread(task, "ThreadDemo04").start();
        System.out.println("===============>"+task.get());
    }
}

    3. Sleep、Yield、Join

    • Sleep：睡眠
    • Yield：让出 cpu，回到就绪状态，有可能又立即回到该线程
    • Join：T1.join(T2)，等 T2 执行完再执行 T1
    public class T02_Sleep_Join_Yield {
    public static void main(String[] args) {
        testJoin();
        testYield();
    }
    private static void testYield() {
        Thread yieldThread01 = new Thread(()->{
            for (int i = 0; i < 100; i++) {
                if(i % 10 == 0){
                    Thread.yield();
                }
                System.out.println("yieldThread01 ======>" + i);
            }
        },"yieldThread01");

        Thread yieldThread02 = new Thread(()->{
            for (int i = 0; i < 100; i++) {
                if(i % 10 == 0){
                    Thread.yield();
                }
                System.out.println("yieldThread02 ======>" + i);
            }
        },"yieldThread02");
        // 发现 yieldThread01、yieldThread02 交互执行
        yieldThread02.start();
        yieldThread01.start();
    }
    private static void testJoin(){
        Thread t = new Thread(() -> {
            for (int i = 0; i < 10; i++) {
                try {
                    TimeUnit.SECONDS.sleep(1);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                System.out.println(i);
            }
        });

        t.start();
        try {
            t.join();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        // 因为 t join 了主线程，需要等 t 执行完
        for (int i = 0; i < 10; i++) {
            try {
                TimeUnit.SECONDS.sleep(1);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            System.out.println("main" + i);
        }
    }
}

    4. ThreadState

    NEW、RUNNABLE、Running、BLOCKED、WAITING、TIMED_WAITING、TERMINATED

    public class T03_Thread_State {
    public static void main(String[] args) throws InterruptedException {
        Thread th = new Thread(() -> {
            for (int i = 0; i < 100; i++) {
                if(i == 20){
                    try {
                        Thread.sleep(2000l);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
                System.out.println(i);
            }
        });
        // 新建：NEW
        System.out.println(th.getState());
        th.start();
        // 就绪：RUNNABLE
        System.out.println(th.getState());
        Thread.sleep(200l);
        // TIMED_WAITING
        System.out.println(th.getState());
        Thread.sleep(2000l);
        // 死亡：TERMINATED
        System.out.println(th.getState());
    }
}

    5. synchronized

    1. 为什么要加锁？两个程序对同一个数字加一，如果不加锁，可能最终结果是1。
    public class T04_Synchronized_02 implements Runnable{
    private int count = 1000;
    // 如果不加 synchronized，可能两个线程同时拿到 1000，然后都减1后，count 值是999，应该是998才对
    @Override
    public /*synchronized*/ void run() {
        synchronized (this){
            count--;
        }
        System.out.println(Thread.currentThread().getName() + "count:" + count);
    }
    public static void main(String[] args) {
        T04_Synchronized_02 T = new T04_Synchronized_02();
        // 启动1000个线程去消费1000个数据。
        for (int i = 0; i < 1000; i++) {
            new Thread(T).start();
        }
    }
}

    1. 对业务写方法加锁，对读业务方法不加锁（会有脏读问题）
    public class T04_Synchronized_04 {
    public static void main(String[] args) {
        Accout accout = new Accout();

    1. 一个同步方法可以调用另外一个同步方法，一个线程已经拥有某个对象的锁，再次申请的时候仍然会得到该对象的锁，也就是说synchronized获得的锁是可重入的
    public class T04_Synchronized_05 {
    public static void main(String[] args) throws InterruptedException {
        new T04_Synchronized_05().m1();
    }
    synchronized void m1() throws InterruptedException {
        System.out.println("m1 start");
        // 调用该对象同步方法(重入)
        m2();
        System.out.println("m1 end");
    }
    synchronized void m2() throws InterruptedException {
        System.out.println("m2 start");
        TimeUnit.SECONDS.sleep(1);
        System.out.println("m2 end");
    }
}

    public class T04_Synchronized_07 {

    }
}

    public class T04_Synchronized_08 {
    /*final*/ Object o = new Object();
    void m(){
        synchronized (o){

    6. synchronized 底层实现

    JDK 早期使用的是重量级锁。后来提出了锁升级的概念。

    锁升级

    • 偏向锁：线程来了之后，不加锁，只是记录线程 ID，认为这个对象是这个线程独有
    • 自旋锁：线程来了之后，发现和记录的线程 ID 不相等，则进行锁升级。锁升级首先尝试进行自旋锁的升级。对象被其他线程锁定时，在旁边转圈，默认转 10 圈，如果还没有拿到就升级重量级锁。线程数量少，线程执行时间短则考虑自旋锁，否则消耗 CPU 太大，应该用重量级锁。
    • 重量级锁：经过 OS，进入等待队列，不再占用 CPU 时间。

    7. volatile

    • 保证线程可见性（MESI）：通过 CPU 的缓存一致性协议实现

      • java 有堆内存（共享内存）和栈内存（线程工作内存），栈内存访问堆内存中值时，每个线程会copy 一个副本到自己的栈内存。那么如果该值被某个线程修改，什么时候写回到共享内存不好控制（应该是修改后立马写回，待确认），并且其他线程什么时候发现该值已经改变了也不好控制。所以一个线程对共享数据的更改，不能及时反应到其他线程，这就叫线程之间的不可见。volatile 可以解决该问题。
    /**
 * volatile 关键字，使一个变量在多个线程间可见
 * A B线程都用到一个变量，java默认是A线程中保留一份copy，这样如果B线程修改了该变量，则A线程未必知道
 *
 * 在下面的代码中，running是存在于堆内存的t对象中
 * 当线程t1开始运行的时候，会把running值从内存中读到t1线程的工作区，在运行过程中直接使用这个copy值，并不会每次都去
 * 读取堆内存，这样，当主线程修改running的值之后，t1线程感知不到，所以不会停止运行
 *
 * 使用volatile，将会强制所有线程都去堆内存中读取running的值
 */
public class T05_Volatile_01 {
    volatile boolean running = true;
    void m(){
        System.out.println("m start");
        while (running){}
        System.out.println("m end");
    }

    public static void main(String[] args) {
        T05_Volatile_01 t = new T05_Volatile_01();
        new Thread(t::m).start();
        try {
            TimeUnit.SECONDS.sleep(1);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        // 主线程修改值，如果不用 volatile，其他线程不可见
        t.running = false;
    }
}

    • 禁止指令重排序（CPU 为了提高效率，可能一个指令没有执行完时就执行下一个指令，这时需要编译器可以对指令进行重新排序）

      • DCL（Double Check Lock） 单例

    • volatile 修饰的引用类型（包括数组）只能保证引用本身的可见性，不能保证内部字段的可见性

    public class T05_Volatile_02 {
    boolean running = true;
    volatile static T05_Volatile_02 t = new T05_Volatile_02();
    void m(){
        System.out.println("m start");
        while (running){}
        System.out.println("m end");
    }
    public static void main(String[] args) {
        new Thread(t::m).start();
        try {
            TimeUnit.SECONDS.sleep(1);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        t.running = false;
    }
}
    public class T05_Volatile_03 {
    private static class Data {
        int a, b;
        public Data(int a, int b) {
            this.a = a;
            this.b = b;
        }
    }
    volatile static Data data;

    public static void main(String[] args) {
        Thread writer = new Thread(()->{
            for (int i = 0; i < 10000; i++) {
                data = new Data(i, i);
            }
        });
        Thread reader = new Thread(()->{
            while (data == null) {
            }
            int x = data.a;
            int y = data.b;
            if(x != y) {
                System.out.printf("a = %s, b=%s%n", x, y);
            }
        });
        reader.start();
        writer.start();
        try {
            reader.join();
            writer.join();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        System.out.println("end");
    }
}
    • volatile 并不能保证多个线程共同修改 running 变量时所带来的不一致问题，也就是说 volatile 不能替代 synchronized，synchronized 可以保证可见性和原子性，volatile 只能保证可见性
    public class T05_Volatile_04 {
    volatile int count = 0;
    void m(){
        for (int i = 0; i < 100000; i++) {
            /*synchronized (this){*/
                count++;
            /*}*/
        }
    }

    8. CAS(自旋锁、无锁优化、乐观锁)

    • Compare And Set

    • cas(V, Excected, NewValue)

      • if V == E 设置新值
      • otherwise try again or fail
      • 例如执行 m=0;m++。cas(0,1){for(;;) 如果当前m值==0，则执行 m=1}

    • CPU 是指令级操作，只有一步原子操作，不可被打断

    • 会有 ABA 问题。解决：1. 忽略，2. 加版本号

 * 解决同样的问题的更高效的方法，使用AtomXXX类

    9.2 LongAdder

    public class T07_LongAdder_01 {
    static int count1 = 0;
    static AtomicLong count2 = new AtomicLong();
    static LongAdder count3 = new LongAdder();
    static Object o = new Object();

    public static void main(String[] args) throws InterruptedException {
        Thread[] threads = new Thread[1000];
        for (int i = 0; i < threads.length; i++) {
            threads[i] = new Thread(()->{
                for (int j = 0; j < 100000; j++) {
                    synchronized (o){
                        count1++;
                    }
                }
            });
        }
        long start = System.currentTimeMillis();
        for (int i = 0; i < threads.length; i++) threads[i].start();
        for (int i = 0; i < threads.length; i++) threads[i].join();
        System.out.println("count1:"+count1+";sync time:" + (System.currentTimeMillis() - start));

        for (int i = 0; i < threads.length; i++) {
            threads[i] = new Thread(()->{
                for (int j = 0; j < 100000; j++) count2.incrementAndGet();
            });

        start = System.currentTimeMillis();
        for (int i = 0; i < threads.length; i++) threads[i].start();
        for (int i = 0; i < threads.length; i++) threads[i].join();
        System.out.println("count2:"+count2+";atomic time:" + (System.currentTimeMillis() - start));
